import java.util.*;
import javax.swing.*;
public class WCC {
    public static void main(String args[]) {
        Scanner br=new Scanner(System.in);
        System.out.println("Enter the string");
        String sen=br.nextLine();
        System.out.println("The given String is: " + sen);
        int tot = 1;
        int i = 0;
        while (i < sen.length()) {
            if ((sen.charAt(i) == ' ') && (sen.charAt(i + 1) != ' '))  {
                tot++;
            }
            i++;
        }
        System.out.println("Number of words in the given string: " +  tot);
    }
}
